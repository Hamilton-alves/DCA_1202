// implementação da função de ordenação bubble sort
int * ordenar(int *vet, int n){
    int aux, i, j;
    for(j=n-1; j>=1; j--){
        for(i=0; i<j; i++){
            if(vet[i]>vet[i+1]){
                aux=vet[i];
                    vet[i]=vet[i+1];
                    vet[i+1]=aux;
            }
        }
    }
    return vet;
}
// Função de comparação do qsort
int compare(const void * a, const void * b) {
  return ( *(int*)a - *(int*)b );
}
