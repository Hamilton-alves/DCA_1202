#include <stdio.h>
#include <stdlib.h>
void mostrar(float *arr, int n) {
    printf("\n[ ");
    for(int i=0; i < n; i++) {
        printf("%f ", arr[i]);
    }
    printf(" ]\n\n");
}
int compare(const void *a, const void *b) {
  return ( *(float*)a - *(float*)b );
}

    void (*ptr_qsrot)(void *arr, size_t n, size_t size, int(*compar)(const void *, const void *));
